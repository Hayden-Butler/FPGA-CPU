Program Counter

//ProgramCounter

module ProgramCounter (
	//inputs
		input logic Clock,
		input logic Reset,
		input logic [15:0] LoadValue,
		input logic signed [8:0] Offset,
		input logic OffsetEnable,
		input logic LoadEnable,
		output logic signed [15:0] CounterValue
	);
	
	always_ff @ (posedge Clock) begin
		
		if (Reset) begin
			CounterValue <= 16'd0;
		end

		else if (LoadEnable) begin
			CounterValue <= LoadValue;
		end
		
		else if (OffsetEnable) begin
			CounterValue <= $signed(CounterValue) + $signed(Offset);
		end
		
		else begin
			CounterValue <= CounterValue + 16'd1;
		end
		
	end
	
endmodule
